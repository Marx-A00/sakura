        val carbs: Int,     // grams
        val fat: Int,       // grams
        val calories: Int
    )

        val reps: Int,

    **`OrgSchema.kt`:**

    Define constants controlling the exact org format. This file IS the format spec — OrgWriter and OrgParser both reference it.

    ```kotlin
    object OrgSchema {
        // Date heading format: * <2026-04-09 Thu>
        val ORG_DATE_FORMATTER: DateTimeFormatter = DateTimeFormatter
            .ofPattern("yyyy-MM-dd EEE", Locale.ENGLISH)

        fun formatDateHeading(date: LocalDate): String =
            "* <${date.format(ORG_DATE_FORMATTER)}>"

        // Date heading regex for parsing
        val DATE_HEADING_REGEX = Regex("""^\* <(\d{4}-\d{2}-\d{2}) \w{3}>$""")
        val DATE_PARSE_FORMATTER: DateTimeFormatter = DateTimeFormatter.ofPattern("yyyy-MM-dd")

        // Meal group heading: ** Breakfast
        val MEAL_LABELS = listOf("Breakfast", "Lunch", "Dinner", "Snacks")
        fun formatMealHeading(label: String): String = "** $label"
        val MEAL_HEADING_REGEX = Regex("""^\*\* (\w+)$""")

        // Exercise group heading: ** Push Day
        fun formatExerciseGroupHeading(label: String): String = "** $label"

        // Food entry: - Chicken and rice  |P: 42g  C: 55g  F: 8g  Cal: 460|
        fun formatFoodEntry(entry: OrgFoodEntry): String =
            "- ${entry.name}  |P: ${entry.protein}g  C: ${entry.carbs}g  F: ${entry.fat}g  Cal: ${entry.calories}|"

        val FOOD_ENTRY_REGEX = Regex(
            """^- (.+?)\s{2}\|P: (\d+)g\s{2}C: (\d+)g\s{2}F: (\d+)g\s{2}Cal: (\d+)\|$"""
        )

        // Exercise entry: - Bench Press  |3x8  80kg|
        fun formatExerciseEntry(entry: OrgExerciseEntry): String =
            "- ${entry.name}  |${entry.sets}x${entry.reps}  ${formatWeight(entry.weight)}${entry.unit}|"

        private fun formatWeight(weight: Double): String =
            if (weight == weight.toLong().toDouble()) weight.toLong().toString()
            else weight.toString()

        val EXERCISE_ENTRY_REGEX = Regex(
            """^- (.+?)\s{2}\|(\d+)x(\d+)\s{2}([\d.]+)(kg|lbs)\|$"""
        )
    }
    ```

    **IMPORTANT: Locale.ENGLISH is mandatory** for the date formatter (see research Pitfall 2). The user's device may be set to Portuguese — without explicit locale, day-of-week abbreviations break Emacs org parsing.

    **Schema design rationale (from research Open Question 1):**
    The inline pipe notation (`|P: 42g  C: 55g  F: 8g  Cal: 460|`) was chosen because:
    1. It passes org-lint (plain text within list items is valid org)
    2. It is compact and human-readable in Emacs
    3. It avoids property drawers (syntactically heavier, not needed for Phase 1)
    4. The double-space + pipe delimiter is unambiguous for regex parsing

    This schema is a proposal. The research notes it must be validated by hand in Emacs with `M-x org-lint` before Plan 02 implements the parser/writer. The validation checkpoint is in Plan 02.
